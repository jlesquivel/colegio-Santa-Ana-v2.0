    Sub llena_deducciones()
        Try
            Dim id_rub As Integer = Me.BindingContext(DsPlanilla1, "empleados").Current.item("id_emp")
            Me.CheckedListBox1.Items.Clear()
            Dim reg, inst As String
            Dim pos As Integer = 0
            Dim dsnuevo As New DataSet
            Dim indi As Integer()
            Dim fila As DataRow
            Dim itemC As Object

            dsnuevo.Tables.Add("a")
            conn.llena(dsnuevo, "a", "select * from rubros")

            ReDim indices(dsnuevo.Tables("a").Rows.Count)
            For Each fila In dsnuevo.Tables("a").Rows
                pos = Me.CheckedListBox1.Items.Add(fila.Item("nombre"))
                indices(pos) = fila.Item("id_rubro")
            Next

            inst = "SELECT empleado_rubro.id_rubro, empleado_rubro.Id_emp, rubros.nombre FROM empleado_rubro INNER JOIN rubros ON empleado_rubro.id_rubro = rubros.id_rubro WHERE (empleado_rubro.Id_emp = " & id_rub & ")"
            dsnuevo.Tables.Add("b")
            conn.llena(dsnuevo, "b", inst)
            For Each fila In dsnuevo.Tables("b").Rows
                pos = Me.CheckedListBox1.Items.IndexOf(fila.Item("nombre"))
                If pos >= 0 Then
                    Me.CheckedListBox1.SetItemChecked(pos, True)
                End If
            Next
        Catch ex As Exception
            MessageBox.Show(ex.Message)
        End Try
    End Sub

    Sub guarda_deducciones()

            inst = "insert empleado_rubro(id_rubro,id_emp) values(" & indices(pos) & "," & Tid_emp & ")"
            conn.ejecuta(inst)
        Next
    End Sub
